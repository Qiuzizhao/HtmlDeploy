import time import random # 游戏初始化数据 money = 50 # 初始金币 crop_list = { 1: {"name": "小麦", "cost": 10, "grow_day": 3, "sell": 30}, 2: {"name": "白菜", "cost": 15, "grow_day": 5, "sell": 50}, 3: {"name": "草莓", "cost": 30, "grow_day": 8, "sell": 120} } field = [] # 田地：[作物编号,剩余生长天数] day = 1 # 当前天数 def show_menu(): print("\n===== 田园种植小游戏 | 第{}天 | 金币：{} =====".format(day, money)) print("1.购买种子播种 2.浇水(加速1天生长，5金币) 3.施肥(加速3天生长，15金币)") print("4.查看田地状态 5.收获成熟作物卖钱 6.结束游戏") def show_crop(): print("\n【种子商店】") for k,v in crop_list.items(): print(f"{k}.{v['name']} | 种子价{v['cost']} | 生长{v['grow_day']}天 | 售价{v['sell']}") while True: show_menu() opt = input("请输入操作序号：") if opt == "1": show_crop() try: cid = int(input("选择要播种的作物编号：")) if cid not in crop_list: print("编号错误！") continue crop = crop_list[cid] if money >= crop["cost"]: money -= crop["cost"] field.append([cid, crop["grow_day"]]) print(f"成功种下{crop['name']}！剩余生长{crop['grow_day']}天") day += 1 else: print("金币不足，买不起种子！") except: print("输入数字！") elif opt == "2": if not field: print("田里没作物，不用浇水！") elif money >=5: money -=5 for i in range(len(field)): if field[i][1]>0: field[i][1]-=1 print("浇水完成，所有作物加速1天！") day +=1 else: print("金币不够浇水！") elif opt == "3": if not field: print("田里没作物！") elif money >=15: money -=15 for i in range(len(field)): if field[i][1]>0: field[i][1] = max(0, field[i][1]-3) print("施肥成功，加速3天！") day +=1 else: print("金币不足！") elif opt == "4": if not field: print("田地空空如也~") else: print("\n【田地详情】") for idx,item in enumerate(field): cname = crop_list[item[0]]['name'] left = item[1] if left <=0: stat = "✅已成熟可收获" else: stat = f"🌱生长中，还需{left}天" print(f"{idx+1}.{cname}:{stat}") elif opt == "5": gain = 0 new_field = [] for item in field: cid,left = item if left <=0: gain += crop_list[cid]["sell"] else: new_field.append(item) field = new_field money += gain if gain>0: print(f"收获完毕，卖出获得{gain}金币！当前金币{money}") day +=1 else: print("暂无成熟作物！") elif opt == "6": print(f"游戏结束！一共玩了{day}天，最终金币：{money}") break else: print("输入1~6有效数字！") time.sleep(0.3)
