import random import os # 清屏函数（适配Windows和Mac/Linux） def clear_screen(): os.system('cls' if os.name == 'nt' else 'clear') # 游戏数据配置 WEAPONS = [ "小刀", "手枪", "冲锋枪", "步枪", "狙击枪" ] LOOTS = [ "医疗包", "止痛药", "子弹", "现金", "高级配件", "加密文档", "金条" ] ENEMIES = [ "游荡者", "精英游荡者", "特遣队员" ] # 玩家状态 player = { "hp": 100, "max_hp": 100, "weapon": "小刀", "backpack": [], "warehouse": [], # 成功撤离后存入仓库 "position": "农场出生点" } # 游戏主菜单 def main_menu(): clear_screen() print("=" * 40) print(" 暗区突围 - 控制台小游戏") print("=" * 40) print(f"当前仓库物资数量：{len(player['warehouse'])} 件") print("1. 开始游戏") print("2. 查看仓库") print("3. 退出游戏") print("=" * 40) choice = input("请选择操作：") if choice == "1": start_game() elif choice == "2": check_warehouse() elif choice == "3": print("感谢游玩！") exit() else: input("输入错误，按回车返回...") main_menu() # 查看仓库 def check_warehouse(): clear_screen() print("=" * 30) print(" 个人仓库") print("=" * 30) if not player["warehouse"]: print("仓库空空如也，快去撤离获取物资吧！") else: for i, item in enumerate(player["warehouse"], 1): print(f"{i}. {item}") input("\n按回车返回主菜单...") main_menu() # 开始游戏 def start_game(): clear_screen() # 重置本局状态 player["hp"] = player["max_hp"] player["backpack"] = [] player["weapon"] = random.choice(["小刀", "手枪"]) player["position"] = "农场平房区" print("=" * 40) print(" 游戏开始！") print(f"当前位置：{player['position']}") print(f"生命值：{player['hp']} 武器：{player['weapon']}") print(f"背包：空") print("=" * 40) game_loop() # 游戏主循环 def game_loop(): while True: print(f"\n【当前位置】{player['position']}") print(f"【状态】HP：{player['hp']} | 武器：{player['weapon']} | 背包物资：{len(player['backpack'])}件") print("1. 移动区域") print("2. 搜索物资") print("3. 休息回血") print("4. 尝试撤离") print("5. 放弃本局") choice = input("请选择行动：") if choice == "1": move() elif choice == "2": search_loot() elif choice == "3": rest() elif choice == "4": evacuate() break elif choice == "5": print("你放弃了本局，背包物资全部丢失！") input("按回车返回主菜单...") main_menu() break else: print("输入错误，请重新选择！") # 随机触发敌人 if random.randint(1, 3) == 1: enemy_encounter() if player["hp"] <= 0: print("你被击败了！背包物资全部丢失！") input("按回车返回主菜单...") main_menu() break # 移动区域 def move(): positions = ["农场平房区", "谷物交易站", "装卸区", "民宅区", "撤离点附近"] player["position"] = random.choice(positions) print(f"你移动到了：{player['position']}") # 搜索物资 def search_loot(): clear_screen() print("正在搜索物资...") loot = random.choice(LOOTS) player["backpack"].append(loot) print(f"你找到了：【{loot}】，已放入背包！") # 小概率获得武器 if random.randint(1, 5) == 1: new_weapon = random.choice(WEAPONS) if WEAPONS.index(new_weapon) > WEAPONS.index(player["weapon"]): player["weapon"] = new_weapon print(f"恭喜！你捡到了更好的武器：{new_weapon}") # 休息回血 def rest(): if player["hp"] >= player["max_hp"]: print("生命值已满，无需休息！") return heal = random.randint(10, 30) player["hp"] = min(player["hp"] + heal, player["max_hp"]) print(f"你休息了一会，恢复了{heal}点生命值！当前HP：{player['hp']}") # 遭遇敌人 def enemy_encounter(): clear_screen() enemy = random.choice(ENEMIES) print(f"\n⚠️ 遭遇敌人！【{enemy}】向你发起攻击！") # 战斗判定 player_power = WEAPONS.index(player["weapon"]) + 1 enemy_power = random.randint(1, 4) if player_power > enemy_power: print(f"你击败了{enemy}！") loot = random.choice(LOOTS) player["backpack"].append(loot) print(f"缴获物资：{loot}") else: damage = random.randint(30, 60) player["hp"] -= damage print(f"战斗失败！受到{damage}点伤害！剩余HP：{player['hp']}") # 撤离 def evacuate(): clear_screen() print("正在前往撤离点...") # 撤离成功率70% if random.randint(1, 10) <= 7: print("✅ 撤离成功！") print(f"本局带回物资：{player['backpack']}") # 物资存入仓库 player["warehouse"].extend(player["backpack"]) else: print("❌ 撤离失败！被埋伏的敌人袭击！") print("背包物资全部丢失！") input("按回车返回主菜单...") main_menu() # 启动游戏 if __name__ == "__main__": main_menu()
